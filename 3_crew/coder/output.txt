# Calculate the first 10,000 terms of the series: 1 - 1/3 + 1/5 - 1/7 + ...
def calculate_series(terms):
    total = 0
    for i in range(terms):
        if i % 2 == 0:
            total += 1 / (2 * i + 1)
        else:
            total -= 1 / (2 * i + 1)
    return total * 4
num_terms = 10000
result = calculate_series(num_terms)
print("Result:", result)
